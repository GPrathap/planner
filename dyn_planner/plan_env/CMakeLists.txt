cmake_minimum_required(VERSION 2.8.3)
project(plan_env)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  rospy
  std_msgs
  edtmap_msg
  visualization_msgs
)


find_package(Eigen3 REQUIRED)
find_package(PCL 1.7 REQUIRED)

catkin_package(
 INCLUDE_DIRS include
 LIBRARIES plan_env
 CATKIN_DEPENDS roscpp std_msgs edtmap_msg
#  DEPENDS system_lib
)

include_directories( 
    SYSTEM 
    include 
    ${catkin_INCLUDE_DIRS}
    ${Eigen3_INCLUDE_DIRS} 
    ${PCL_INCLUDE_DIRS}
)

link_directories(${PCL_LIBRARY_DIRS})

set(CMAKE_CXX_FLAGS "-std=c++11 ${CMAKE_CXX_FLAGS} -O3 -Wall")

add_library( plan_env  src/edtmap_wrapper.cpp)

target_link_libraries( plan_env ${catkin_LIBRARIES} ${PCL_LIBRARIES})  

add_executable(edt_map_builder_node src/edt_map_builder_node.cpp src/sdf_map.cpp)
target_link_libraries(edt_map_builder_node ${catkin_LIBRARIES} ${PCL_LIBRARIES})
